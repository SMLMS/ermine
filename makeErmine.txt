# Define required macros here
CXXFLAGS = -Wall -fPIC -std=c++11
CXX = g++
INCLUDES = -I$(HOME)/Dokumente/Code/ermine
LIBS = -L$(HOME)/Dokumente/Code/ermine/source
LFLAGS := -shared

BOOSTLIBS = -I/usr/include
BOOSTSOURCE = -L/usr/include/boost
BOOSTFLAG = -lboost_program_options
BOOSTFLAG += -lboost_system
BOOSTFLAG += -lboost_filesystem

SOURCES = main.cpp
SOURCES += source/ermineStatement.cpp
SOURCES += source/ermineParser.cpp
SOURCES += source/ermineExceptions.cpp
SOURCES += source/smlmsFolder.cpp
SOURCES += source/ermineFilenames.cpp
OBJECTS = $(SOURCES: .cpp=.o)



EXECUTABLE := prog/ermine

all:$(OBJECTS) $(EXECUTABLE)

$(EXECUTABLE): $(OBJECTS)
	$(CXX) $? $(BOOSTLIBS) $(BOOSTSOURCE) $(BOOSTFLAG) $(LIBS) $(INCLUDES) -o $@

.cpp.o:
	$(CXX) $(CXXFLAGS) $(INCLUDES) -c $<

clean:
	-rm *.o prog/ermine
